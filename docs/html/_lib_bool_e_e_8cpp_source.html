19 std::vector<std::string> LibBoolEE::singleParse(const std::string & formula, const char op) {
20  int start_pos = -1;
21  int parity_count = 0;
22  std::vector<std::string> subexpressions;
23  for (int i = 0; i < static_cast<int>(formula.size()); i++) {
24  if (formula[i] == ')') {
25  parity_count--;
26  }
27  else if (formula[i] == '(') {
28  parity_count++;
29  if (start_pos == -1) {
30  start_pos = i;
31  }
32  }
33  else if (parity_count == 0) {
34  if (start_pos == -1) {
35  if (belongsToName(formula[i]) || formula[i] == '!') {
36  start_pos = i;
37  }
38  }
39  else if (!(belongsToName(formula[i]) || formula[i] == '!')) {
40  if (op == formula[i]) {
41  subexpressions.push_back(formula.substr(start_pos, i - start_pos));
42  start_pos = i+1;
43  }
44  else if (formula[i] != '&' && formula[i] != '|') {
45  throw std::runtime_error("Unknown operator '" + std::string(1, formula[i]) + "' in the (sub)expression '" + formula + "'.");
46  }
47  }
48  }
49  }
50  if (start_pos != -1) {
51  subexpressions.push_back(formula.substr(start_pos, formula.size() - start_pos));
52  }
53  if (parity_count != 0) {
54  throw std::runtime_error("Wrong parenthesis parity in the (sub)expression '" + formula + "'.");
55  }
56  return subexpressions;
57 }

63 bool LibBoolEE::resolveRec(const std::string &source, const Vals & valuation) {
64  if (source.empty()) {
65  throw std::runtime_error("An empty subexpression was encountered");
66  }
67 
68  std::string formula = source;
69  char current_op = '|';
70  // Try to divide by |
71  std::vector<std::string> subexpressions = singleParse(source, current_op);
72  // No | on the top level
73  if (subexpressions.size() == 1) {
74  current_op = '&';
75  subexpressions = singleParse(source, current_op);
76  }
77 
78  // No valid name found
79  if (subexpressions.size() == 0) {
80  throw std::runtime_error("The subexpression " + source + " is not a valid formula.");
81  }
82  // No binary top level operator found
83  else if (subexpressions.size() == 1) {
84  if (source[0] == '!') {
85  return !resolve(source.substr(1), valuation);
86  }
87  else if (source[0] == '(') {
88  return resolve(source.substr(1, source.size() - 2), valuation);
89  }
90  else if (source == "1") {
91  return true;
92  }
93  else if (source == "0") {
94  return false;
95  }
96  else if (valuation.count(source) == 0) {
97  throw std::runtime_error("Variable '" + source + "' not found in the interpretation.");
98  }
99  else {
100  return valuation.at(source);
101  }
102  }
103  else {
104  if (current_op == '|') {
105  bool result = false;
106  for (std::vector<std::string>::iterator it = subexpressions.begin(); it != subexpressions.end(); it++) {
107  result |= resolve(*it, valuation);
108  }
109  return result;
110  }
111  else { // The operator was set to &
112  bool result = true;
113  for (std::vector<std::string>::iterator it = subexpressions.begin(); it != subexpressions.end(); it++) {
114  result &= resolve(*it, valuation);
115  }
116  return result;
117  }
118  }
119 }
