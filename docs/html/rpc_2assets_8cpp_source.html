41  size_t msglen = message.length();

43  if (msglen == 64 && !AreMessagingDeployed()) {

45  }
46  } else {

50 
51  bool fNotIPFS = false;

53  fNotIPFS = true;

57 
58  if (fNotIPFS && !IsHex(message))
59  throw JSONRPCError(RPC_INVALID_PARAMS, std::string("Invalid IPFS/Txid hash"));
