22 std::string ValueFromAmountString(const CAmount& amount, const int8_t units)
23 {
24  bool sign = amount < 0;
25  int64_t n_abs = (sign ? -amount : amount);
26  int64_t quotient = n_abs / COIN;
27  int64_t remainder = n_abs % COIN;
28  remainder = remainder / pow(10, 8 - units);
29 
30  if (units == 0 && remainder == 0) {
31  return strprintf("%s%d", sign ? "-" : "", quotient);
32  }
33  else {
34  return strprintf("%s%d.%0" + std::to_string(units) + "d", sign ? "-" : "", quotient, remainder);
35  }
36 }

48 std::string FormatScript(const CScript& script)
49 {
50  std::string ret;
51  CScript::const_iterator it = script.begin();
52  opcodetype op;
53  while (it != script.end()) {
54  CScript::const_iterator it2 = it;
55  std::vector<unsigned char> vch;
56  if (script.GetOp2(it, op, &vch)) {
57  if (op == OP_0) {
58  ret += "0 ";
59  continue;
60  } else if ((op >= OP_1 && op <= OP_16) || op == OP_1NEGATE) {
61  ret += strprintf("%i ", op - OP_1NEGATE - 1);
62  continue;
63  } else if (op >= OP_NOP && op <= OP_NOP10) {
64  std::string str(GetOpName(op));
65  if (str.substr(0, 3) == std::string("OP_")) {
66  ret += str.substr(3, std::string::npos) + " ";
67  continue;
68  }
69  }
70  if (vch.size() > 0) {
71  ret += strprintf("0x%x 0x%x ", HexStr(it2, it - vch.size()), HexStr(it - vch.size(), it));
72  } else {
73  ret += strprintf("0x%x ", HexStr(it2, it));
74  }
75  continue;
76  }
77  ret += strprintf("0x%x ", HexStr(it2, script.end()));
78  break;
79  }
80  return ret.substr(0, ret.size() - 1);
81 }

99 std::string ScriptToAsmStr(const CScript& script, const bool fAttemptSighashDecode)
100 {
101  std::string str;
102  opcodetype opcode;
103  std::vector<unsigned char> vch;
104  CScript::const_iterator pc = script.begin();
105  while (pc < script.end()) {
106  if (!str.empty()) {
107  str += " ";
108  }
109  if (!script.GetOp(pc, opcode, vch)) {
110  str += "[error]";
111  return str;
112  }
113  if (0 <= opcode && opcode <= OP_PUSHDATA4) {
114  if (vch.size() <= static_cast<std::vector<unsigned char>::size_type>(4)) {
115  str += strprintf("%d", CScriptNum(vch, false).getint());
116  } else {
117  // the IsUnspendable check makes sure not to try to decode OP_RETURN data that may match the format of a signature
118  if (fAttemptSighashDecode && !script.IsUnspendable()) {
119  std::string strSigHashDecode;
120  // goal: only attempt to decode a defined sighash type from data that looks like a signature within a scriptSig.
121  // this won't decode correctly formatted public keys in Pubkey or Multisig scripts due to
122  // the restrictions on the pubkey formats (see IsCompressedOrUncompressedPubKey) being incongruous with the
123  // checks in CheckSignatureEncoding.
124  if (CheckSignatureEncoding(vch, SCRIPT_VERIFY_STRICTENC, nullptr)) {
125  const unsigned char chSigHashType = vch.back();
126  if (mapSigHashTypes.count(chSigHashType)) {
127  strSigHashDecode = "[" + mapSigHashTypes.find(chSigHashType)->second + "]";
128  vch.pop_back(); // remove the sighash type byte. it will be replaced by the decode.
129  }
130  }
131  str += HexStr(vch) + strSigHashDecode;
132  } else {
133  str += HexStr(vch);
134  }
135  }
136  } else {
137  str += GetOpName(opcode);
138  }
139  }
140  return str;
141 }

150 void ScriptPubKeyToUniv(const CScript& scriptPubKey,
151  UniValue& out, bool fIncludeHex)
152 {
153  txnouttype type;
154  std::vector<CTxDestination> addresses;
155  int nRequired;
156 
157  out.pushKV("asm", ScriptToAsmStr(scriptPubKey));
158  if (fIncludeHex)
159  out.pushKV("hex", HexStr(scriptPubKey.begin(), scriptPubKey.end()));
160 
161  if (!ExtractDestinations(scriptPubKey, type, addresses, nRequired)) {
162  out.pushKV("type", GetTxnOutputType(type));
163  return;
164  }
165 
166  out.pushKV("reqSigs", nRequired);
167  out.pushKV("type", GetTxnOutputType(type));
168 
170  if (type == TX_NEW_ASSET || type == TX_TRANSFER_ASSET || type == TX_REISSUE_ASSET) {
171  UniValue assetInfo(UniValue::VOBJ);
172 
173  std::string name;
174  CAmount amount;
175  std::string _assetAddress;
176 
177  if (GetAssetInfoFromScript(scriptPubKey, name, amount)) {
178  assetInfo.pushKV("name", name);
179  assetInfo.pushKV("amount", ValueFromAmount(amount));
180 
181  switch (type) {
182  case TX_NEW_ASSET:
183  if (IsAssetNameAnOwner(name)) {
184  // pwnd n00b
185  } else {
186  CNewAsset asset;
187  if (AssetFromScript(scriptPubKey, asset, _assetAddress)) {
188  assetInfo.pushKV("units", asset.units);
189  assetInfo.pushKV("reissuable", asset.nReissuable > 0 ? true : false);
190  if (asset.nHasIPFS > 0) {
191  assetInfo.pushKV("ipfs_hash", EncodeAssetData(asset.strIPFSHash));
192  }
193  }
194  }
195  break;
196  case TX_TRANSFER_ASSET:
197  break;
198  case TX_REISSUE_ASSET:
199  CReissueAsset asset;
200  if (ReissueAssetFromScript(scriptPubKey, asset, _assetAddress)) {
201  if (asset.nUnits >= 0) {
202  assetInfo.pushKV("units", asset.nUnits);
203  }
204  assetInfo.pushKV("reissuable", asset.nReissuable > 0 ? true : false);
205  if (!asset.strIPFSHash.empty()) {
206  assetInfo.pushKV("ipfs_hash", EncodeAssetData(asset.strIPFSHash));
207  }
208  }
209  break;
210  }
211  }
212 
213  out.pushKV("asset", assetInfo);
214  }
215 
216  if (type == TX_RESTRICTED_ASSET_DATA) {
217  UniValue assetInfo(UniValue::VOBJ);
218  CNullAssetTxData data;
219  CNullAssetTxVerifierString verifierData;
220  std::string address;
221  if (AssetNullDataFromScript(scriptPubKey, data, address)) {
222  AssetType type;
223  IsAssetNameValid(data.asset_name, type);
224  if (type == AssetType::QUALIFIER || type == AssetType::SUB_QUALIFIER) {
225  assetInfo.pushKV("asset_name", data.asset_name);
226  assetInfo.pushKV("qualifier_type", data.flag ? "adding qualifier" : "removing qualifier");
227  assetInfo.pushKV("address", address);
228  } else if (type == AssetType::RESTRICTED) {
229  assetInfo.pushKV("asset_name", data.asset_name);
230  assetInfo.pushKV("restricted_type", data.flag ? "freezing address" : "unfreezing address");
231  assetInfo.pushKV("address", address);
232  }
233  } else if (GlobalAssetNullDataFromScript(scriptPubKey, data)) {
234  assetInfo.pushKV("restricted_name", data.asset_name);
235  assetInfo.pushKV("restricted_type", data.flag ? "freezing" : "unfreezing");
236  assetInfo.pushKV("address", "all addresses");
237  } else if (AssetNullVerifierDataFromScript(scriptPubKey, verifierData)) {
238  assetInfo.pushKV("verifier_string", verifierData.verifier_string);
239  }
240 
241  out.pushKV("asset_data", assetInfo);
242  }
245  UniValue a(UniValue::VARR);
246  for (const CTxDestination& addr : addresses) {
247  a.push_back(EncodeDestination(addr));
248  }
249  out.pushKV("addresses", a);
250 }

252 void TxToUniv(const CTransaction& tx, const uint256& hashBlock, UniValue& entry, bool include_hex, int serialize_flags)
253 {
254  entry.pushKV("txid", tx.GetHash().GetHex());
255  entry.pushKV("hash", tx.GetWitnessHash().GetHex());
256  entry.pushKV("version", tx.nVersion);
257  entry.pushKV("size", (int)::GetSerializeSize(tx, SER_NETWORK, PROTOCOL_VERSION));
258  entry.pushKV("vsize", (GetTransactionWeight(tx) + WITNESS_SCALE_FACTOR - 1) / WITNESS_SCALE_FACTOR);
259  entry.pushKV("locktime", (int64_t)tx.nLockTime);
260 
261  UniValue vin(UniValue::VARR);
262  for (unsigned int i = 0; i < tx.vin.size(); i++) {
263  const CTxIn& txin = tx.vin[i];
264  UniValue in(UniValue::VOBJ);
265  if (tx.IsCoinBase())
266  in.pushKV("coinbase", HexStr(txin.scriptSig.begin(), txin.scriptSig.end()));
267  else {
268  in.pushKV("txid", txin.prevout.hash.GetHex());
269  in.pushKV("vout", (int64_t)txin.prevout.n);
270  UniValue o(UniValue::VOBJ);
271  o.pushKV("asm", ScriptToAsmStr(txin.scriptSig, true));
272  o.pushKV("hex", HexStr(txin.scriptSig.begin(), txin.scriptSig.end()));
273  in.pushKV("scriptSig", o);
274  if (!tx.vin[i].scriptWitness.IsNull()) {
275  UniValue txinwitness(UniValue::VARR);
276  for (const auto& item : tx.vin[i].scriptWitness.stack) {
277  txinwitness.push_back(HexStr(item.begin(), item.end()));
278  }
279  in.pushKV("txinwitness", txinwitness);
280  }
281  }
282  in.pushKV("sequence", (int64_t)txin.nSequence);
283  vin.push_back(in);
284  }
285  entry.pushKV("vin", vin);
286 
287  UniValue vout(UniValue::VARR);
288  for (unsigned int i = 0; i < tx.vout.size(); i++) {
289  const CTxOut& txout = tx.vout[i];
290 
291  UniValue out(UniValue::VOBJ);
292 
293  out.pushKV("value", ValueFromAmount(txout.nValue));
294  out.pushKV("n", (int64_t)i);
295 
296  UniValue o(UniValue::VOBJ);
297  ScriptPubKeyToUniv(txout.scriptPubKey, o, true);
298  out.pushKV("scriptPubKey", o);
299  vout.push_back(out);
300  }
301  entry.pushKV("vout", vout);
302 
303  if (!hashBlock.IsNull())
304  entry.pushKV("blockhash", hashBlock.GetHex());
305 
306  if (include_hex) {
307  entry.pushKV("hex", EncodeHexTx(tx, serialize_flags)); // the hex-encoded transaction. used the name "hex" to be consistent with the verbose output of "getrawtransaction".
308  }
309 }
